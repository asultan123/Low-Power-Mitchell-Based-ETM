1%%%%EM4
 2% Approximate 8X8 Array Multiplier based on AMA4
 3function C=AMA4ArrayMultiplier_8X8(A,B)
 4% C = A X B
 5%convert input from decimal to 4 bits binary
 6
 7Abin=de2bi(A,8);
 8Bbin=de2bi(B,8);
 9PP=zeros(8,8);   %(R,C)
10for i=1:8    %% row   generate partial products  
11    for j=1:8   % column
12    PP(i,j)=and(Abin(j),Bbin(i));  % inner loop   PP(R,C)
13    end
14end
15S0=PP(1,1) ;
16[C22, S1]= HalfAdder(PP(2,1),PP(1,2));
17[C23, S22]= AMA4FA(PP(1,3),PP(2,2),C22);
18[C24, S23]= AMA4FA(PP(1,4),PP(2,3),C23);
19[C25, S24]= AMA4FA(PP(1,5),PP(2,4),C24);
20[C26, S25]= AMA4FA(PP(1,6),PP(2,5),C25);
21[C27, S26]= AMA4FA(PP(1,7),PP(2,6),C26);
22[C28, S27]= AMA4FA(PP(1,8),PP(2,7),C27);
23[C29, S28]= HalfAdder(PP(2,8),C28);    % first row of FAs
24
25[C32, S2]= HalfAdder(PP(3,1),S22);
26[C33, S32]=AMA4FA(PP(3,2),S23,C32);
27[C34 ,S33]=AMA4FA(PP(3,3),S24,C33);
28[C35, S34]=AMA4FA(PP(3,4),S25,C34);
29[C36, S35]=AMA4FA(PP(3,5),S26,C35);
30[C37 ,S36]=AMA4FA(PP(3,6),S27,C36);
31[C38, S37]=MAFA(PP(3,7),S28,C37);
32[C39, S38]=MAFA(PP(3,8),C29,C38);   % second row of FAS
33
34[C42, S3]= HalfAdder(PP(4,1),S32);
35[C43, S42]=AMA4FA(PP(4,2),S33,C42);
36[C44 ,S43]=AMA4FA(PP(4,3),S34,C43);
37[C45, S44]=AMA4FA(PP(4,4),S35,C44);
38[C46, S45]=AMA4FA(PP(4,5),S36,C45);
39[C47 ,S46]=MAFA(PP(4,6),S37,C46);
40[C48, S47]=MAFA(PP(4,7),S38,C47);
41[C49, S48]=MAFA(PP(4,8),C39,C48);   % Third row of FAS
42
43
44[C52, S4]= HalfAdder(PP(5,1),S42);
45[C53, S52]=AMA4FA(PP(5,2),S43,C52);
46[C54 ,S53]=AMA4FA(PP(5,3),S44,C53);
47[C55, S54]=AMA4FA(PP(5,4),S45,C54);
48[C56, S55]=MAFA(PP(5,5),S46,C55);
49[C57 ,S56]=MAFA(PP(5,6),S47,C56);
50[C58, S57]=MAFA(PP(5,7),S48,C57);
51[C59, S58]=MAFA(PP(5,8),C49,C58);   % Fourth row of FA
52
53[C62,S5] = HalfAdder(PP(6,1),S52);
54[C63,S62]=AMA4FA(PP(6,2),S53,C62);
55[C64,S63]=AMA4FA(PP(6,3),S54,C63);
56[C65,S64]=MAFA(PP(6,4),S55,C64);
57[C66,S65]=MAFA(PP(6,5),S56,C65);
58[C67,S66]=MAFA(PP(6,6),S57,C66);
59[C68,S67]=MAFA(PP(6,7),S58,C67);
60[C69,S68]=MAFA(PP(6,8),C59,C68);   % 5th row of FA
61
62[C72,S6]= HalfAdder(PP(7,1),S62);
63[C73,S72]=AMA4FA(PP(7,2),S63,C72);  
64[C74,S73]=MAFA(PP(7,3),S64,C73);
65[C75,S74]=MAFA(PP(7,4),S65,C74);
66[C76,S75]=MAFA(PP(7,5),S66,C75);
67[C77,S76]=MAFA(PP(7,6),S67,C76);
68[C78,S77]=MAFA(PP(7,7),S68,C77);
69[C79,S78]=MAFA(PP(7,8),C69,C78);   % siXth row of FA
70
71[C82,S7] =HalfAdder(PP(8,1),S72);
72[C83,S8] =MAFA(PP(8,2),S73,C82);
73[C84,S9] =MAFA(PP(8,3),S74,C83);
74[C85,S10]=MAFA(PP(8,4),S75,C84);
75[C86,S11]=MAFA(PP(8,5),S76,C85);
76[C87,S12]=MAFA(PP(8,6),S77,C86);
77[C88,S13]=MAFA(PP(8,7),S78,C87);
78[S15,S14]=MAFA(PP(8,8),C79,C88);   % 7th row of FA
79%************************************%
80output=[S0 S1 S2 S3 S4 S5 S6 S7 S8 S9 S10 S11 S12 S13 S14 S15];
81C=bi2de(output);
82
83end
